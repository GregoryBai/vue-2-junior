<template>
  <Navbar :navbar-text="getPageName" />
  <div class="wrapper" v-if="isRootDir"><router-view /></div>
  <router-view v-else />
</template>

<script>
import Navbar from "./components/Navbar"
import { computed } from "vue"
import { useRoute } from "vue-router"
export default {
  setup() {
    const route = useRoute()

    const isRootDir = computed(() => route.path === "/")
    const getPageName = computed(() =>
      route.path === "/" ? "LeadHit" : `Leadhit: ${route.path.slice(1)}`
    )

    return { route, isRootDir, getPageName }
  },

  components: {
    Navbar,
  },
}
</script>
